web与httpServer接口定义

url 功能说明 入参 出参 权限控制
/api 跳转到主页 - html -
/api/signup 跳转到用户注册页面 - html -
/api/signin 跳转到用户登录页面 - html -
/api/info 跳转到用户信息页面 - html -
/api/updatePage 跳转到更新信息页面 - html -
/api/createUser Func:新建用户 username,password,nickname,profilepicture json(resFlag,username,token,resMsg) -
/api/login Func:登录 username,password json -
/api/getToken Func:获取token - json(token) -
/api/private/update Func:更新信息 username,password,nickname,profilepicture json 有
/api/private/queryUser Func:查询用户信息 username json 有

type User struct {
    Username string `xorm:"varchar(100)"`

    Nickname string `xorm:"varchar(200)"`
    Picuri   string `xorm:"varchar(200)"`
}

    string username = 1;    // 用户名
    string password = 2;    // 密码
    string nickname = 3;    // 昵称
    string picuri   = 4;    // 图片uri
    string resFlag  = 5;    // 消息flag
    string resMsg   = 6;    // 消息内容
}

mysql数据库

## 安装mysql
brew install mysql

## 启动mqsql服务
brew services start mysql

## 连接mysql(注:刚安装时是没密码,先手动设置一些密码后退出再重连)
mysql -u root -p root   (用户 root 密码 root)


#### 以下操作在 terminal 的 mysql 交互界面中操作
## 创建 项目db
CREATE DATABASE IF NOT EXISTS entry_task_db;

## 使用新建的db
use entry_task_db;

## 创建 项目表
CREATE TABLE IF NOT EXISTS `user`(
`username`      VARCHAR(100) NOT NULL,
`password`      VARCHAR(100) NOT NULL,
`nickname`      VARCHAR(200),
`picuri`        VARCHAR(200),
PRIMARY KEY ( `username` )
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

压测

环境和软件准备:

  1. 安装 lua brew install lua
  2. 安装 wrk brew install wrk

压测脚本在项目根目录下的 /lua 中

  1. login 压测 :线程8, 连接数200, 压测时间30s, 打印延迟统计信息, 指定lua脚本(不同的用户)

    wrk -t8 -c200 -d30s --latency -s entry_task_login.lua http://localhost:9999/api/login

## entry_task_login.lua 内容:
wrk.method="POST"
-- wrk.body='{"username": "1","password": "1"}'
wrk.headers["Content-Type"]="application/x-www-form-urlencoded"
request = function ()
local tmp = math.random(1, 10000000)
local body = ""
    .. string.format("&username=%d", tmp)
    .. string.format("&password=%d", tmp)
return wrk.format('POST', nil, nil, body)
end
